function startReplay(gameName) {
    try {
        if (botState.replayRecording) stopReplay();
        room.startRecording();
        botState.replayRecording = true;
        botState.replayGameName = gameName;
        console.log('🎥 Replay iniciado: ' + gameName);
    } catch(e) { console.log('⚠️ Replay start error: ' + e.message); }
}

function stopReplay() {
    if (!botState.replayRecording) return;
    try {
        var data = room.stopRecording();
        botState.replayRecording = false;
        if (data && data.length > 0) {
            var arr = Array.from(data);
            console.log('🎥 Replay detenido: ' + botState.replayGameName + ' (' + arr.length + ' bytes)');
            if (typeof window.__sendReplay === 'function') {
                window.__sendReplay(arr, botState.replayGameName).catch(function(e) {
                    console.log('⚠️ Replay send error: ' + e);
                });
            }
        }
    } catch(e) {
        botState.replayRecording = false;
        console.log('⚠️ Replay stop error: ' + e.message);
    }
}

function getRankByWins(wins) {
    for (var i = 0; i < RANK_TIERS.length; i++) {
        if (wins >= RANK_TIERS[i].min) {
            return RANK_TIERS[i];
        }
    }
    return RANK_TIERS[RANK_TIERS.length - 1];
}

// Clear bans pero mantener los permanentes
function safeClearBans() {
    try { room.clearBans(); } catch(e){}
    var players = room.getPlayerList();
    var now = Date.now();
    for (var i = 0; i < players.length; i++) {
        var pAuth = players[i].auth || (botState.authMap && botState.authMap[players[i].id]);
        if (!pAuth) continue;
        // Re-kickear bans permanentes
        if (PERMA_BANNED_AUTHS.indexOf(pAuth) !== -1) {
            try { room.kickPlayer(players[i].id, 'Ban permanente', true); } catch(e){}
            continue;
        }
        // Re-kickear temp bans activos
        if (botState.tempBans[pAuth] && botState.tempBans[pAuth].until > now) {
            var secs = Math.ceil((botState.tempBans[pAuth].until - now) / 1000);
            try { room.kickPlayer(players[i].id, 'Ban temporal - ' + secs + 's restantes', true); } catch(e){}
        }
    }
}

// SISTEMA ANTI-SPAM
// ============================================
var SPAM_CONFIG = {
    limit: 8,        // mensajes máximos antes de mutear
    window: 6000,    // ventana de tiempo (ms) para contar mensajes
    muteBase: 60000  // duración base del mute (ms): 1 min, 2 min, 3 min...
};

function checkSpam(player) {
    var auth = getAuth(player);
    if (!auth) return false; // sin auth, no se aplica spam
    // CREADORES nunca son muteados
    if (CREATOR_AUTHS.indexOf(auth) !== -1) return false;

    var now = Date.now();
    if (!botState.spamMessages[auth]) botState.spamMessages[auth] = [];

    // Filtrar mensajes fuera de la ventana
    botState.spamMessages[auth] = botState.spamMessages[auth].filter(function(t) {
        return now - t < SPAM_CONFIG.window;
    });
    botState.spamMessages[auth].push(now);

    // Advertencia antes del mute (al llegar a 6 mensajes)
    if (botState.spamMessages[auth].length === SPAM_CONFIG.limit - 2) {
        room.sendAnnouncement('⚠️ ' + player.name + ', estás enviando mensajes muy rápido. Si continúas serás muteado.', player.id, 0xFFFF00, 'bold');
    }

    if (botState.spamMessages[auth].length >= SPAM_CONFIG.limit) {
        // Calcular duración del mute (1 min * veces muteado)
        var prevCount = (botState.spamMutes[auth] ? botState.spamMutes[auth].count : 0);
        var muteCount = prevCount + 1;
        var muteDuration = SPAM_CONFIG.muteBase * muteCount;

        botState.spamMutes[auth] = { until: now + muteDuration, count: muteCount };
        botState.spamMessages[auth] = []; // resetear contador

        var minutos = Math.round(muteDuration / 60000);
        room.sendAnnouncement(
            '🔇 ' + player.name + ' fue muteado por ' + minutos + ' minuto(s) por spam.',
            null, 0xFF6600, 'bold'
        );
        room.sendAnnouncement(
            '🔇 Estás muteado por ' + minutos + ' minuto(s) por enviar demasiados mensajes.',
            player.id, 0xFF0000, 'bold'
        );

        // Timer para avisarle cuando se desmutea
        if (botState.spamMuteTimers[auth]) clearTimeout(botState.spamMuteTimers[auth]);
        botState.spamMuteTimers[auth] = setTimeout(function() {
            delete botState.spamMutes[auth];
            delete botState.spamMuteTimers[auth];
            // Buscar al jugador por auth para notificarle
            var players = room.getPlayerList();
            for (var i = 0; i < players.length; i++) {
                if (getAuth(players[i]) === auth) {
                    room.sendAnnouncement('🔊 Ya puedes chatear de nuevo.', players[i].id, 0x00FF00, 'bold');
                    break;
                }
            }
        }, muteDuration);

        return true; // está muteado
    }
    return false;
}

function isSpamMuted(player) {
    var auth = getAuth(player);
    if (!auth) return false;
    if (!botState.spamMutes[auth]) return false;
    return Date.now() < botState.spamMutes[auth].until;
}

// Formatear dinero con separador de miles (estilo $50.000)
function formatMoney(amount) {
    return "$" + amount.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ".");
}

// Buffer de chatlog: acumula mensajes y envía cada 5s para evitar 429
function flushChatlog() {
    if (botState.chatlogBuffer.length === 0) return;
    var batch = botState.chatlogBuffer.splice(0, 20); // max 20 por envío
    var text = batch.join('\n');
    if (text.length > 1900) text = text.substring(0, 1900) + '...';
    if (typeof window.__discordWebhook === 'function') {
        window.__discordWebhook('chatlog', text);
    }
    // Si quedan más, programar otro flush
    if (botState.chatlogBuffer.length > 0) {
        botState.chatlogTimer = setTimeout(flushChatlog, 5000);
    } else {
        botState.chatlogTimer = null;
    }
}

function bufferChatlog(line) {
    botState.chatlogBuffer.push(line);
    if (!botState.chatlogTimer) {
        botState.chatlogTimer = setTimeout(flushChatlog, 5000);
    }
}

// Helper para compras de tienda: verifica saldo, cobra, aplica cooldown
function shopBuy(playerAuth, playerName, playerId, cost, commandName, cooldownMs, callback) {
    if (!playerAuth) { room.sendAnnouncement('⚠️ No se pudo verificar tu identidad', playerId, 0xFF6600); return; }
    var now = Date.now();
    if (cooldownMs > 0) {
        if (!botState.shopCooldowns[playerAuth]) botState.shopCooldowns[playerAuth] = {};
        var last = botState.shopCooldowns[playerAuth][commandName] || 0;
        if (now - last < cooldownMs) {
            var wait = Math.ceil((cooldownMs - (now - last)) / 1000);
            room.sendAnnouncement('⏳ ' + commandName + ' en cooldown. Espera ' + wait + 's', playerId, 0xFF6600);
            return;
        }
    }
    if (typeof window.__dbGetStats !== 'function') { room.sendAnnouncement('⚠️ Tienda no disponible.', playerId, 0xFF6600); return; }
    window.__dbGetStats(playerAuth).then(function(stats) {
        var balance = (stats && stats.balance) ? stats.balance : 0;
        if (balance < cost) {
            room.sendAnnouncement('⚠️ Saldo insuficiente. Necesitas ' + formatMoney(cost) + ' (tienes ' + formatMoney(balance) + ')', playerId, 0xFF6600);
            return;
        }
        try { window.__dbAddBalance(playerAuth, playerName, -cost); } catch(e) {}
        if (cooldownMs > 0) botState.shopCooldowns[playerAuth][commandName] = now;
        callback();
    }).catch(function() { room.sendAnnouncement('❌ Error al verificar saldo.', playerId, 0xFF0000); });
}

function startAutoAnnouncements() {
    setInterval(function() {
        // No enviar anuncios si hay un juego activo (minijuego en curso)
        if (botState.gameActive || botState.chatBlocked) {
            // Marcar anuncio pendiente para enviarlo cuando termine la explicación/minijuego
            botState.pendingAnnouncement = true;
            return; // No enviar anuncios durante juegos activos o cuando el chat está bloqueado
        }
        
        botState.chatBlocked = true;
        
        room.sendAnnouncement(
            "━━━━━━━━━━━━━━━━━━━━━━━━\n" +
            "💬 ÚNETE A NUESTRO DISCORD\n" +
            "🔗 https://discord.gg/YvFHkSZUvE\n" +
            "━━━━━━━━━━━━━━━━━━━━━━━━",
            null,
            0x5865F2,
            "bold",
            2
        );
        
        setTimeout(function() {
            botState.chatBlocked = false;
        }, 2000);
    }, 120000); // Cada 2 minutos
}

// Vigilar anuncios pendientes y enviarlos cuando sea seguro
function startPendingAnnouncementWatcher() {
    setInterval(function() {
        if (botState.pendingAnnouncement && !botState.gameActive && !botState.chatBlocked) {
            botState.pendingAnnouncement = false;
            botState.chatBlocked = true;
            room.sendAnnouncement(
                "━━━━━━━━━━━━━━━━━━━━━━━━\n" +
                "💬 ÚNETE A NUESTRO DISCORD\n" +
                "🔗 https://discord.gg/YvFHkSZUvE\n" +
                "━━━━━━━━━━━━━━━━━━━━━━━━",
                null,
                0x5865F2,
                "bold",
                2
            );
            setTimeout(function() { botState.chatBlocked = false; }, 2000);
        }
    }, 2000);
}

// Manejar kicks/bans realizados desde la UI (click en jugador)
// Firma: onPlayerKicked(kickedPlayer, reason, ban, byPlayer)
room.onPlayerKicked = function(kickedPlayer, reason, ban, byPlayer) {
    try {
        if (byPlayer && byPlayer.id !== 0) {
            var byAuth = getAuth(byPlayer);
            var isCreator = byAuth && CREATOR_AUTHS.indexOf(byAuth) !== -1;
            // Temp admins NO pueden banear NI kickear
            var isTempAdmin = botState.tempAdmins.indexOf(byPlayer.id) !== -1;
            if (isTempAdmin && !isCreator) {
                if (ban) {
                    safeClearBans();
                }
                try { room.setPlayerAdmin(byPlayer.id, false); } catch(e){}
                var taIdx = botState.tempAdmins.indexOf(byPlayer.id);
                if (taIdx !== -1) botState.tempAdmins.splice(taIdx, 1);
                room.sendAnnouncement('⛔ ' + byPlayer.name + ' abusó del admin temporal. Admin removido.', null, 0xFF0000, 'bold');
                return;
            }

            // permitir a creadores banear
            if (ban) {
                if (!isCreator) {
                    // revertir ban: limpiar bans, quitar admin
                    safeClearBans();
                    try { room.setPlayerAdmin(byPlayer.id, false); } catch(e){}
                    room.sendAnnouncement('⛔ ' + byPlayer.name + ' intentó banear desde la interfaz. Ban revertido y admin removido.', null, 0xFF0000, 'bold');
                }
                return;
            }

            // Si es kick (no ban): aplicar cooldown de 2 minutos por admin (creadores exentos)
            var now = Date.now();
            botState.lastKickBy = botState.lastKickBy || {};
            var last = botState.lastKickBy[byPlayer.id] || 0;
            if (now - last < 120000 && !isCreator) {
                // Violó cooldown: revocar admin temporal y avisar
                try { room.setPlayerAdmin(byPlayer.id, false); } catch(e){}
                room.sendAnnouncement('⛔ Has kickeado demasiado seguido. Tu admin ha sido revocado temporalmente.', byPlayer.id, 0xFF0000, 'bold');
            } else {
                botState.lastKickBy[byPlayer.id] = now;
            }
        }
    } catch (e) { console.error('[ROOM] onPlayerKicked error', e); }
};

function startIMPOSTOR() {
    botState.gameActive = true;

    // Guardar y quitar admin a admins temporales durante el juego
    var savedTempAdmins = botState.tempAdmins.slice();
    for (var ta = 0; ta < savedTempAdmins.length; ta++) {
        var taId = savedTempAdmins[ta];
        try { room.setPlayerAdmin(taId, false); } catch(e) {}
        room.sendAnnouncement('⚠️ Tu admin ha sido removido temporalmente durante IMPOSTOR. Se te devolverá al terminar.', taId, 0xFFFF00, 'bold');
    }

    if (typeof IMPOSTOR !== 'undefined') {
        IMPOSTOR.start(room, function(result) {
            // Restaurar admin a los temp admins que sigan conectados
            for (var ra = 0; ra < savedTempAdmins.length; ra++) {
                var raId = savedTempAdmins[ra];
                var raPlayer = room.getPlayer(raId);
                if (raPlayer && botState.tempAdmins.indexOf(raId) !== -1) {
                    try { room.setPlayerAdmin(raId, true); } catch(e) {}
                    room.sendAnnouncement('👑 Tu admin temporal ha sido restaurado.', raId, 0xFFD700, 'bold');
                }
            }

            if (result && result.type === 'inocentes_win') {
                // Dar $2.000 a cada inocente vivo
                for (var i = 0; i < result.inocenteIds.length; i++) {
                    var pAuth = botState.authMap[result.inocenteIds[i]];
                    if (pAuth && typeof window.__dbAddBalance === 'function') {
                        try { window.__dbAddBalance(pAuth, result.inocenteNames[i], 2000); } catch(e) {}
                    }
                }
            } else if (result && result.type === 'impostor_win') {
                // Dar premio a los impostores (array)
                var impPrize = (result.impostorIds && result.impostorIds.length === 2) ? 5000 : 10000;
                if (result.impostorIds) {
                    for (var ii = 0; ii < result.impostorIds.length; ii++) {
                        var impAuth = botState.authMap[result.impostorIds[ii]];
                        var impName = result.impostorNames ? result.impostorNames[ii] : 'Impostor';
                        if (impAuth && typeof window.__dbAddBalance === 'function') {
                            try { window.__dbAddBalance(impAuth, impName, impPrize); } catch(e) {}
                        }
                    }
                }
            } else if (result && result.type === 'bufon_win') {
                // Dar premio al bufon ($10.000)
                var bufonAuth = botState.authMap[result.bufonId];
                if (bufonAuth && typeof window.__dbAddBalance === 'function') {
                    try { window.__dbAddBalance(bufonAuth, result.bufonName, 10000); } catch(e) {}
                }
            }
            botState.gameActive = false;
            setTimeout(checkAndStart, 5000);
        });
    } else {
        // Restaurar admins si el módulo no existe
        for (var rb = 0; rb < savedTempAdmins.length; rb++) {
            var rbPlayer = room.getPlayer(savedTempAdmins[rb]);
            if (rbPlayer) {
                try { room.setPlayerAdmin(savedTempAdmins[rb], true); } catch(e) {}
            }
        }
        botState.gameActive = false;
        setTimeout(checkAndStart, 5000);
    }
}

// Ban temporal robusto: trackea por auth con timestamp de expiración
function applyTempBan(playerId, duration) {
    var auth = botState.authMap[playerId] || null;
    var player = room.getPlayer(playerId);
    var playerName = player ? player.name : '#' + playerId;

    // Trackear por auth
    if (auth) {
        botState.tempBans[auth] = { until: Date.now() + duration * 1000, name: playerName };
    }

    // Limpiar timer anterior si existe
    if (botState.banTimers[playerId]) clearTimeout(botState.banTimers[playerId]);

    // Agregar a lista legacy de baneados
    if (botState.bannedPlayers.indexOf(playerId) === -1) botState.bannedPlayers.push(playerId);

    // Marcar como kickeado por el juego para que anti-dodge no lo castigue
    if (auth) botState.kickedByGame[auth] = true;

    room.kickPlayer(playerId, 'Ban temporal por ' + duration + 's', true);
    console.log('⚔️ Ban temporal a ' + playerName + ' (auth: ' + auth + ') por ' + duration + 's');

    botState.banTimers[playerId] = setTimeout(function() {
        // Remover de lista legacy
        var index = botState.bannedPlayers.indexOf(playerId);
        if (index !== -1) botState.bannedPlayers.splice(index, 1);
        // Remover de tempBans por auth
        if (auth && botState.tempBans[auth]) delete botState.tempBans[auth];

        safeClearBans();
        room.sendAnnouncement('✅ Ban temporal removido - ' + playerName + ' puede volver a entrar', null, 0x00FF00, 'bold');
        delete botState.banTimers[playerId];
        console.log('✅ Ban temporal removido para ' + playerName);
    }, duration * 1000);
}

// Limpieza periódica de temp bans expirados (cada 15s)
setInterval(function() {
    var now = Date.now();
    var expired = false;
    for (var auth in botState.tempBans) {
        if (botState.tempBans[auth].until <= now) {
            console.log('🔓 Limpieza periódica: ban expirado para ' + botState.tempBans[auth].name);
            delete botState.tempBans[auth];
            expired = true;
        }
    }
    if (expired) safeClearBans();
}, 15000);

// JUICIO PÚBLICO (Ban Vote)
// ============================================
function startBanVote(room, data) {
    // data = { playerId, playerName, auth, duration }
    botState.banVoteActive = true;
    botState.gameActive = true;

    try { room.stopGame(); } catch(e) {}
    try { room.setCustomStadium(mapBanVote); } catch(e) {}

    // Poner acusado en rojo, todos los demás en azul
    var allPlayers = room.getPlayerList().filter(function(p) { return p.id !== 0; });
    var accusedPresent = false;
    allPlayers.forEach(function(p) {
        if (p.id === data.playerId) {
            try { room.setPlayerTeam(p.id, 1); } catch(e) {} // Rojo
            accusedPresent = true;
        } else {
            try { room.setPlayerTeam(p.id, 2); } catch(e) {} // Azul
        }
    });

    // Si el acusado no está, cancelar juicio
    if (!accusedPresent) {
        room.sendAnnouncement('⚖️ El acusado ' + data.playerName + ' se desconectó. Juicio cancelado.', null, 0xFF6600, 'bold');
        botState.banVoteActive = false;
        botState.gameActive = false;
        setTimeout(checkAndStart, 3000);
        return;
    }

    try { room.startGame(); } catch(e) {}
    try { room.pauseGame(true); } catch(e) {}
    botState.chatBlocked = true;

    // Anunciar instrucciones durante la pausa
    room.sendAnnouncement(
        '⚖️ ¡JUICIO PÚBLICO! ⚖️\n' +
        '🔴 Acusado: ' + data.playerName + '\n' +
        '⬆️ Zona ARRIBA = SÍ merece ban\n' +
        '⬇️ Zona ABAJO = NO merece ban\n' +
        '⏱️ Comienza en 5s...',
        null, 0xFFFF00, 'bold', 2
    );

    // Tracking de votos: { playerId: 'ban' | 'noban' }
    var votes = {};
    var voteNames = {};

    // Después de 5s despausar y empezar votación
    botState.banVoteTimeout = setTimeout(function() {
        botState.chatBlocked = false;
        try { room.pauseGame(false); } catch(e) {}
        room.sendAnnouncement('⏱️ ¡15 segundos para votar! Muevan su jugador a la zona.', null, 0xFFFF00, 'bold', 2);

        // Check posiciones cada 500ms
        var checkInterval = setInterval(function() {
            var players = room.getPlayerList().filter(function(p) { return p.id !== 0 && p.team === 2; });
            players.forEach(function(p) {
                if (!p.position) return;
                var x = p.position.x;
                var y = p.position.y;

                // Zona "SÍ merece ban": x:65-381, y:-6 a 108
                if (x >= 65 && x <= 381 && y >= -6 && y <= 108) {
                    if (votes[p.id] !== 'ban') {
                        votes[p.id] = 'ban';
                        voteNames[p.id] = p.name;
                        room.sendAnnouncement('✅ ' + p.name + ' votó: SÍ merece ban', null, 0xFF0000, 'bold');
                    }
                }
                // Zona "NO merece ban": x:67-381, y:-174 a -52
                else if (x >= 67 && x <= 381 && y >= -174 && y <= -52) {
                    if (votes[p.id] !== 'noban') {
                        votes[p.id] = 'noban';
                        voteNames[p.id] = p.name;
                        room.sendAnnouncement('❌ ' + p.name + ' votó: NO merece ban', null, 0x00FF00, 'bold');
                    }
                }
            });
        }, 500);

        // Guardar referencia para limpieza
        botState.banVoteInterval = checkInterval;

        // A los 15 segundos, contar votos
        botState.banVoteTimeout = setTimeout(function() {
        clearInterval(checkInterval);
        botState.banVoteInterval = null;

        var banCount = 0;
        var noBanCount = 0;
        for (var pid in votes) {
            if (votes[pid] === 'ban') banCount++;
            else if (votes[pid] === 'noban') noBanCount++;
        }

        room.sendAnnouncement('📊 Resultado: SÍ ban: ' + banCount + ' | NO ban: ' + noBanCount, null, 0xFFFF00, 'bold');

        var verdict;
        if (banCount > noBanCount) {
            verdict = 'ban';
        } else if (noBanCount > banCount) {
            verdict = 'noban';
        } else {
            // Empate o sin votos → aleatorio
            verdict = Math.random() < 0.5 ? 'ban' : 'noban';
            if (banCount === 0 && noBanCount === 0) {
                room.sendAnnouncement('⚖️ ¡Nadie votó! Decidiendo al azar...', null, 0xFFFF00, 'bold');
            } else {
                room.sendAnnouncement('⚖️ ¡EMPATE! Decidiendo al azar...', null, 0xFFFF00, 'bold');
            }
        }

        setTimeout(function() {
            if (verdict === 'ban') {
                room.sendAnnouncement('⚖️ VEREDICTO: ¡CULPABLE! ' + data.playerName + ' será baneado ' + data.duration + 's', null, 0xFF0000, 'bold', 2);
                // Verificar que sigue conectado
                var accused = room.getPlayer(data.playerId);
                if (accused) {
                    applyTempBan(data.playerId, data.duration);
                } else if (data.auth) {
                    // Si se desconectó, guardar ban por auth
                    botState.tempBans[data.auth] = { until: Date.now() + data.duration * 1000, name: data.playerName };
                }
            } else {
                room.sendAnnouncement('⚖️ VEREDICTO: ¡INOCENTE! ' + data.playerName + ' se salva del ban', null, 0x00FF00, 'bold', 2);
            }

            try { room.stopGame(); } catch(e) {}
            botState.banVoteActive = false;
            botState.gameActive = false;
            setTimeout(checkAndStart, 3000);
        }, 2000);
        }, 15000);
    }, 5000);
}

function loadLuckyMap(winner) {
    // Check robo de lucky (!robarlucky) - solo roba el Lucky, no la victoria
    var luckyPlayer = winner;
    if (botState.robarLucky) {
        var thief = room.getPlayer(botState.robarLucky.playerId);
        if (thief) {
            room.sendAnnouncement('🏴‍☠️ ' + botState.robarLucky.name + ' le ROBÓ el Lucky a ' + winner.name + '!', null, 0xFF0000, 'bold', 2);
            luckyPlayer = thief;
        }
        botState.robarLucky = null;
    }

    // Mensaje de victoria aleatorio
    var vMsg = victoryMessages[Math.floor(Math.random() * victoryMessages.length)];
    room.sendAnnouncement(vMsg, null, 0xFFD700, "bold");

    // Obtener auth de forma confiable (auth solo disponible en onPlayerJoin, usar authMap)
    var winnerAuth = (botState.authMap && botState.authMap[winner.id])
        ? botState.authMap[winner.id] : winner.auth;

    // Guardar victoria en DB y actualizar cache de rangos
    if (winnerAuth && typeof window.__dbSaveWin === 'function') {
        try { window.__dbSaveWin(winnerAuth, winner.name, botState.currentMinigame || 'UNKNOWN'); } catch(e) {}
        var oldWins = botState.winsCache[winnerAuth] || 0;
        var newWins = oldWins + 1;
        botState.winsCache[winnerAuth] = newWins;
        // Verificar si subió de rango
        var oldRank = getRankByWins(oldWins);
        var newRank = getRankByWins(newWins);
        if (oldRank.label !== newRank.label) {
            room.sendAnnouncement(
                "🎖️ " + winner.name + " SUBIO DE RANGO!\n" +
                oldRank.label + " → " + newRank.label,
                null, newRank.color, "bold", 2
            );
        }
    }

    // Sistema de rachas
    for (var auth in botState.winStreaks) {
        if (auth !== winnerAuth) {
            delete botState.winStreaks[auth];
        }
    }
    botState.winStreaks[winnerAuth] = (botState.winStreaks[winnerAuth] || 0) + 1;
    var streak = botState.winStreaks[winnerAuth];
    if (streak >= 2) {
        room.sendAnnouncement("🔥 " + winner.name + " lleva " + streak + " VICTORIAS SEGUIDAS!", null, 0xFF4500, "bold", 2);
    }

    // Guardar mejor racha en DB
    if (winnerAuth && typeof window.__dbSaveBestStreak === 'function') {
        try { window.__dbSaveBestStreak(winnerAuth, streak); } catch(e) {}
    }

    // Sistema monetario: recompensa por ganar
    var reward = moneyConfig.winReward;
    var streakBonus = (streak >= 2) ? (streak - 1) * moneyConfig.streakBonus : 0;
    var totalReward = reward + streakBonus;
    // Check !doble: duplicar recompensa si el ganador tiene apuesta activa
    var dobleUsado = false;
    if (winnerAuth && botState.dobleActive[winnerAuth]) {
        totalReward = totalReward * 2;
        dobleUsado = true;
        delete botState.dobleActive[winnerAuth];
    }

    }

    if (dobleUsado) {
        rewardMsg += " (x2 DOBLE!)";
    }
    if (eventoMulti) rewardMsg += eventoMulti;

    for (var dAuth in botState.dobleActive) {
        if (dAuth !== winnerAuth) delete botState.dobleActive[dAuth];
    }

    room.setPlayerTeam(luckyPlayer.id, 1);

    var allPlayers = room.getPlayerList().filter(p => p.id !== 0);
    allPlayers.forEach(p => {
        if (p.id !== luckyPlayer.id) {
            room.setPlayerTeam(p.id, 0);
        }

                        botState.protectionActive[pAuth] = true;

                        try { window.__dbAddBalance(pAuth, playerName, amount); } catch(e) {}

                    // Guardar por AUTH para que no esquiven saliendo/entrando
                    var spAuth = botState.authMap[playerId] || null;
                    if (!spAuth) return;
                    botState.spectatorNext[spAuth] = true;
                    botState.spectatorLocked[spAuth] = true;
                    try { room.sendAnnouncement('👻 Has sido seleccionado para NO jugar el siguiente minijuego', playerId, 0xFF6600, 'bold'); } catch(e){}

                    if (botState.spectatorLockedTimers[spAuth]) clearTimeout(botState.spectatorLockedTimers[spAuth]);

                        delete botState.spectatorLocked[spAuth];
                        delete botState.spectatorLockedTimers[spAuth];
                        console.log('🔓 Desbloqueo automático de spectatorLocked para auth=' + spAuth);
                    }, 90000);
                },
                onGayTag: function(playerAuth) {
                    // Asignar etiqueta GAY por 4 minijuegos
                    botState.playerTags[playerAuth] = { type: 'GAY', remaining: 4 };
                    console.log("🌈 Etiqueta GAY asignada a auth: " + playerAuth);
                },
                onBanTemp: function(playerId, duration) {
                    // Check protección antes de banear
                    var pBanAuth = botState.authMap[playerId] || null;
                    if (pBanAuth && botState.protectionActive[pBanAuth]) {
                        delete botState.protectionActive[pBanAuth];

                        room.sendAnnouncement('🛡️ PROTECCIÓN ACTIVADA! Ban cancelado para ' + pBanName, null, 0x0000FF, 'bold', 2);
                        return;
                    }
                    // 10% probabilidad de juicio público
                    if (Math.random() < 0.10) {
                        var pName = (room.getPlayer(playerId) || {name: '?'}).name;
                        botState.pendingBanVote = { playerId: playerId, playerName: pName, auth: pBanAuth, duration: duration };
                        room.sendAnnouncement('⚖️ ¡Se activó un JUICIO PÚBLICO para ' + pName + '!', null, 0xFFFF00, 'bold', 2);
                        return;
                    }
                    applyTempBan(playerId, duration);
                },
                isAfk: function(playerId) {
                    var pAuth = botState.authMap[playerId] || null;
                    return !!(pAuth && botState.afkProtected[pAuth]);
                },
                onLuckyPass: function(giverId, receiverId) {
                    var giverAuth = botState.authMap[giverId] || null;
                    var receiverAuth = botState.authMap[receiverId] || null;
                    if (!giverAuth || !receiverAuth) return;
                    var now = Date.now();
                    // Limpiar entradas antiguas (>10 min)
                    botState.luckyPassLog = botState.luckyPassLog.filter(function(e) { return now - e.ts < 600000; });
                    // Buscar pase recíproco reciente: ¿receiver le pasó a giver antes?
                    var pairKey = [giverAuth, receiverAuth].sort().join('|');
                    var reciprocal = false;
                    for (var lp = 0; lp < botState.luckyPassLog.length; lp++) {
                        var entry = botState.luckyPassLog[lp];
                        if (entry.giverAuth === receiverAuth && entry.receiverAuth === giverAuth) {
                            reciprocal = true; break;
                        }
                    }
                    // Registrar este pase
                    botState.luckyPassLog.push({ giverAuth: giverAuth, receiverAuth: receiverAuth, ts: now });
                    if (reciprocal) {
                        var giverName = (room.getPlayer(giverId) || {name:'?'}).name;
                        var receiverName = (room.getPlayer(receiverId) || {name:'?'}).name;
                        if (botState.luckyPassWarned[pairKey]) {
                            // Ya fueron advertidos → ban a ambos por 5 minutos
                            room.sendAnnouncement('🚫 ABUSO DETECTADO: ' + giverName + ' y ' + receiverName + ' se están pasando el Lucky entre sí. BAN 5 min para ambos!', null, 0xFF0000, 'bold', 2);
                            applyTempBan(giverId, 300);
                            applyTempBan(receiverId, 300);
                            delete botState.luckyPassWarned[pairKey];
                            // Limpiar log de este par
                            botState.luckyPassLog = botState.luckyPassLog.filter(function(e) {
                                return !((e.giverAuth === giverAuth && e.receiverAuth === receiverAuth) || (e.giverAuth === receiverAuth && e.receiverAuth === giverAuth));
                            });
                        } else {
                            // Primera vez → advertencia
                            botState.luckyPassWarned[pairKey] = true;
                            room.sendAnnouncement('⚠️ ADVERTENCIA: ' + giverName + ' y ' + receiverName + ' se están pasando el Lucky entre sí. Si lo hacen de nuevo, ambos serán baneados 5 minutos!', null, 0xFFFF00, 'bold', 2);
                        }
                    }
                }
            });
        } catch (e) {
            console.log("❌ Error al cargar mapa Lucky: " + e.message);
            botState.gameActive = false;
            setTimeout(checkAndStart, 5000);
        }
    }, 2000);
}

var _soloInterval = null;

function startSoloEntertainment() {
    if (_soloInterval) return; // ya activo
    _soloMsgIndex = 0;
    // Mezclar mensajes
    for (var i = soloMessages.length - 1; i > 0; i--) {
        var j = Math.floor(Math.random() * (i + 1));
        var tmp = soloMessages[i];
        soloMessages[i] = soloMessages[j];
        soloMessages[j] = tmp;
    }

    _soloInterval = setInterval(function() {
        var players = room.getPlayerList().filter(function(p) { return p.id !== 0; });
        if (players.length !== 1) {
            stopSoloEntertainment();
            return;
        }
        var p = players[0];
        var pAuth = getAuth(p);
        var wins = (pAuth && botState.winsCache[pAuth]) ? botState.winsCache[pAuth] : 0;
        var rank = getRankByWins(wins);

        // Alternar entre mensaje random y stats personales
        if (_soloMsgIndex % 3 === 0 && pAuth) {
            // Cada 3 mensajes, mostrar stats personales
            room.sendAnnouncement(
                "📊 " + p.name + " | " + rank.label + " | 🏆 " + wins + " victorias\n" +
                "⏳ Esperando rival... ¡Invita a alguien!",
                null, rank.color, "bold"
            );
        } else {
            var msg = soloMessages[_soloMsgIndex % soloMessages.length];
            room.sendAnnouncement(msg, null, 0xFFFF00);
        }
        _soloMsgIndex++;
    }, 15000);
}

function stopSoloEntertainment() {
    if (_soloInterval) {
        clearInterval(_soloInterval);
        _soloInterval = null;
    }
}

room.onPlayerJoin = function(player) {
    console.log("👤 " + player.name + " entró | Auth: " + player.auth);

    // Guardar auth del jugador (auth solo está disponible en onPlayerJoin)
    if (player.auth) {
        // Ban permanente
        if (PERMA_BANNED_AUTHS.indexOf(player.auth) !== -1) {
            room.kickPlayer(player.id, 'Ban permanente', true);
            return;
        }

        // Anti-multijoin: si ya hay alguien con el mismo auth, kickear al nuevo
        var existingIds = Object.keys(botState.authMap);
        for (var i = 0; i < existingIds.length; i++) {
            if (botState.authMap[existingIds[i]] === player.auth && existingIds[i] != player.id) {
                room.kickPlayer(player.id, 'Ya estás conectado en esta sala', false);
                console.log('🚫 Anti-multijoin: ' + player.name + ' (auth: ' + player.auth + ') ya conectado como ID ' + existingIds[i]);
                return;
            }
        }

        botState.authMap[player.id] = player.auth;

        // Cargar victorias de DB para el sistema de rangos
        if (typeof window.__dbGetStats === 'function') {
            window.__dbGetStats(player.auth).then(function(stats) {
                if (stats && stats.wins) {
                    botState.winsCache[player.auth] = stats.wins;
                }
            }).catch(function() {});
        }
    }

    // ============================================
    // TRACKING DE TOKENS EN DISCORD
    // ============================================
    if (player.auth && typeof window.__discordWebhook === 'function') {
        var known = botState.knownPlayers[player.auth];
        if (!known) {
            // Jugador nuevo: enviar token a Discord
            window.__discordWebhook('tokens',
                '🆕 **Nuevo jugador**\n' +
                '👤 Nombre: **' + player.name + '**\n' +
                '🔑 Auth: `' + player.auth + '`'
            );
            botState.knownPlayers[player.auth] = { name: player.name, notified: true };
        } else if (known.name !== player.name) {
            // Cambio de nombre: notificar en Discord

                '🔑 Auth: `' + player.auth + '`'
            );
            botState.knownPlayers[player.auth].name = player.name;
        }
    }

    // ============================================
    // ANTI-DODGE: Banear a dodgers de Lucky
    // ============================================
    if (player.auth && botState.luckyDodgers[player.auth]) {
        var dodge = botState.luckyDodgers[player.auth];
        var remaining = dodge.until - Date.now();
        if (remaining > 0) {
            var secs = Math.ceil(remaining / 1000);
            room.sendAnnouncement('⛔ ' + player.name + ' fue baneado por salirse del Lucky para evitar castigo (' + secs + 's restantes)', null, 0xFF0000, 'bold');

            // Programar desban
            setTimeout(function() {
                delete botState.luckyDodgers[player.auth];
                safeClearBans();
                room.sendAnnouncement('✅ Ban por dodge removido - el jugador puede volver', null, 0x00FF00);
            }, remaining);
            return;
        } else {
            delete botState.luckyDodgers[player.auth];
        }
    }

    // ============================================
    // CHECK TEMP BANS POR AUTH (robusto)
    // ============================================
    if (player.auth && botState.tempBans[player.auth]) {
        var tbRemaining = botState.tempBans[player.auth].until - Date.now();
        if (tbRemaining > 0) {

            delete botState.tempBans[player.auth];
            safeClearBans();
        }

    // ============================================
    if (CREATOR_AUTHS.indexOf(player.auth) !== -1) {

    // KING: dar admin automático

        LUCKY_DIOS.onPlayerJoin(room, player);
    }

            "⏳ Hay un juego en curso. Espera a que termine para jugar.",
            player.id,
            0xFFFF00
        );
        return;
    }
    
    if (!botState.startTimeout) {
        var playerCount = room.getPlayerList().filter(p => p.id !== 0).length;
        if (playerCount >= 2) {
            stopSoloEntertainment();
            room.sendAnnouncement(
                "✅ Hay suficientes jugadores!\n⏱️ El juego comenzará en unos segundos...",
                null,
                0x00FF00,
                "bold"
            );
            botState.startTimeout = setTimeout(checkAndStart, 2000);
        } else if (playerCount === 1) {
            startSoloEntertainment();
        }
    }
};

        if (SURVIVOR_VOL17 && SURVIVOR_VOL17.isActive && SURVIVOR_VOL17.isActive()) {
            SURVIVOR_VOL17.stop(room);
        }
        if (WIND && WIND.isActive && WIND.isActive()) {
            WIND.stop(room);
        }
        if (IMPOSTOR && IMPOSTOR.isActive && IMPOSTOR.isActive()) {
            IMPOSTOR.stop(room);
        }

        botState.gameActive = false;
        if (botState.startTimeout) {
            clearTimeout(botState.startTimeout);
            botState.startTimeout = null;
        }
        
        // Reiniciar búsqueda de minijuego cuando haya 2 jugadores
        room.sendAnnouncement(
            "⏳ Esperando más jugadores...",
            null,
            0x00FF00,
            "normal"
        );

        // Iniciar entretenimiento si queda 1 solo
        var soloPlayers = room.getPlayerList().filter(function(p) { return p.id !== 0; }).length;
        if (soloPlayers === 1) {
            startSoloEntertainment();
        }

        // Verificar cada 5 segundos si hay suficientes jugadores
        var checkInterval = setInterval(function() {
            var currentPlayers = room.getPlayerList().filter(p => p.id !== 0).length;
            if (currentPlayers >= 2) {
                clearInterval(checkInterval);
                stopSoloEntertainment();
                checkAndStart();
            }
        }, 5000);
    }
};

room.onPlayerTeamChange = function(changedPlayer, byPlayer) {
    // AFK protection: si el jugador activó !afk, mantenerlo en espectador
    var afkAuth = getAuth(changedPlayer);
    if (afkAuth && botState.afkProtected[afkAuth] && changedPlayer.team !== 0) {
        try { room.setPlayerTeam(changedPlayer.id, 0); } catch(e){}
        try { room.sendAnnouncement('💤 Estás en modo AFK. Escribe !afk para desactivarlo.', changedPlayer.id, 0xFFFF00); } catch(e){}
        return;
    }

    // Bloqueo general de cambios de equipo (por ejemplo durante pausas/instrucciones)
    if (botState.teamLock) {
        // revertir al equipo anterior (si existe)
        try { room.setPlayerTeam(changedPlayer.id, 0); } catch(e){}
        try { room.sendAnnouncement('⛔ Los cambios de equipo están bloqueados ahora mismo.', changedPlayer.id, 0xFF6600); } catch(e){}
        return;
    }

    // Si el jugador está bloqueado como "spectatorLocked" (por auth), forzar espectador
    var cpAuth = getAuth(changedPlayer);
    if (cpAuth && botState.spectatorLocked[cpAuth]) {
        if (changedPlayer.team !== 0) {
            try { room.setPlayerTeam(changedPlayer.id, 0); } catch(e){}
            try { room.sendAnnouncement('👻 No puedes unirte al equipo: estás bloqueado para el siguiente minijuego', changedPlayer.id, 0xFF6600, 'bold'); } catch(e){}
            // Notificar a todos los módulos activos para que quiten al jugador de su lista interna
            var _mods = [LALALA, SURVIVAL, METEORS, JUMPING, WEBSURVIVAL, GALACTIC, GYM, MULTIBALLS,
                DODGEBALL, SUPERMAN, ULTRABALL, NUMBERCHAIRS, SURVIVAL_SQ, COLLISION_TEAM_RACING,
                SPOT_BONK, CHAIRMIX, COVID19, GYMMIX, TRIVIA, GOT, SPACE_CASE, SPACE_MELEE,
                AMERICAN_CARRYBALL, RING_SURVIVAL, BONK_ARENA, BONK_BERU,
                ANIMALCHAIRS_WHALE, BONK_BERU_HAX, COVID2, DOMINIC_SURVIVOR, AIR_HOCKEY,
                FIT_IN_TREN, FITANIC, FORTUNATE_RACE, KAFA_TOPU, KAFA_TOPU_SHADOW,
                KURU_KAFA_BONK, LASER_RACE, IMPOSTOR];
            _mods.forEach(function(m) {
                try { if (m && m.isActive && m.isActive() && m.onPlayerLeave) m.onPlayerLeave(room, changedPlayer); } catch(e){}
            });
        }
        return;
    }

    // Evitar que alguien se mueva al equipo azul (equipo 2) SOLO DURANTE LUCKY o LUCKY HELL
    if ((LUCKY.isActive() || LUCKY_HELL.isActive() || LUCKY_DIOS.isActive()) && changedPlayer.team === 2) {
        room.setPlayerTeam(changedPlayer.id, 0);
        room.sendAnnouncement(
            "⚠️ No puedes unirte al equipo azul durante Lucky",
            changedPlayer.id,
            0xFF6600
        );
        return;
    }
    
    // Si está en spectatorNext (por auth), forzar a espectador
    if (cpAuth && botState.spectatorNext[cpAuth] && changedPlayer.team !== 0) {
        room.setPlayerTeam(changedPlayer.id, 0);
        room.sendAnnouncement(
            "👻 Debes esperar el próximo minijuego",
            changedPlayer.id,
            0xFF6600,
            "bold"
        );
    }
};

room.onPlayerChat = function(player, message) {
    // Escribir en chat cuenta como actividad (evita kick AFK mientras chateas)
    if (player.id !== 0) botState.afkLastActivity[player.id] = Date.now();

    // Resolver auth (no disponible en onPlayerChat, usar authMap)
    var playerAuth = getAuth(player);

    // Log de chat a Discord (solo jugadores, no bot) - con buffer para evitar 429
    if (player.id !== 0) {
        var isCmd = message.charAt(0) === '!';
        var logLine = (isCmd ? '⚡ ' : '') + '**' + player.name + '** (#' + player.id + '): ' + message;
        bufferChatlog(logLine);
    }

    // Bloqueo de chat durante anuncios automáticos
    if (botState.chatBlocked) {
        return false;
    }

    // Anti-spam: verificar si ya está muteado
    if (isSpamMuted(player)) {
        var mute = botState.spamMutes[getAuth(player)];
        var segsLeft = Math.ceil((mute.until - Date.now()) / 1000);
        room.sendAnnouncement('🔇 Muteado. Quedan ' + segsLeft + 's.', player.id, 0xFF0000);
        return false;
    }
    // Anti-spam: contar mensajes y mutear si supera el límite
    if (checkSpam(player)) {
        return false;
    }

    // Anti-links: solo permitir el invite de Discord de la sala
    if (playerAuth && CREATOR_AUTHS.indexOf(playerAuth) === -1) {
        var linkRegex = /https?:\/\/[^\s]+|www\.[^\s]+|discord\.gg\/[^\s]+|bit\.ly\/[^\s]+|t\.me\/[^\s]+/i;
        if (linkRegex.test(message)) {
            // Permitir solo el link del Discord de la sala
            if (message.indexOf('discord.gg/YvFHkSZUvE') === -1) {
                room.sendAnnouncement('⛔ No se permiten links. Solo puedes compartir el Discord: discord.gg/YvFHkSZUvE', player.id, 0xFF0000, 'bold');
                return false;
            }
        }
    }

    // Control de admins temporales - no pueden usar comandos de admin
    // EXCEPCION: Los CREADORES siempre pueden usar kick/ban/set aunque tengan admin temporal
    if (botState.tempAdmins.indexOf(player.id) !== -1 && CREATOR_AUTHS.indexOf(playerAuth) === -1) {
        if (message.startsWith('!kick') || message.startsWith('!ban') || message.startsWith('!set')) {
            room.sendAnnouncement(
                "⛔ Los admins temporales NO pueden kickear, banear ni cambiar mapas",
                player.id,
                0xFF0000,
                "bold"
            );
            room.sendAnnouncement(
                "⚠️ " + player.name + " intentó usar comandos de admin - KICKEADO",
                null,
                0xFF6600,
                "bold"
            );
            setTimeout(function() {
                room.kickPlayer(player.id, "Uso indebido de admin temporal", false);
            }, 1000);
            return false;
        }
    }

    // ============================================
    // COMANDO ADMIN: !mute #ID minutos
    // Solo CREADORES pueden usar este comando
    // ============================================
    if (message.startsWith('!mute')) {
        if (!playerAuth || CREATOR_AUTHS.indexOf(playerAuth) === -1) {
            room.sendAnnouncement('⛔ Solo los CREADORES pueden usar !mute', player.id, 0xFF0000);
            return false;
        }
        var muteParts = message.split(' ');
        if (muteParts.length < 3) {
            room.sendAnnouncement('⚠️ Usa: !mute #ID minutos\nEjemplo: !mute #3 5', player.id, 0xFFFF00);
            return false;
        }
        var muteTargetId = parseInt(muteParts[1].replace('#', ''));
        var muteMinutes = parseInt(muteParts[2]);
        if (isNaN(muteTargetId)) { room.sendAnnouncement('⚠️ ID inválido', player.id, 0xFF6600); return false; }
        if (isNaN(muteMinutes) || muteMinutes < 1) { room.sendAnnouncement('⚠️ Minutos inválidos (mínimo 1)', player.id, 0xFF6600); return false; }
        var muteTarget = room.getPlayer(muteTargetId);
        if (!muteTarget) { room.sendAnnouncement('⚠️ Jugador #' + muteTargetId + ' no encontrado', player.id, 0xFF6600); return false; }
        var muteTargetAuth = getAuth(muteTarget);
        if (!muteTargetAuth) { room.sendAnnouncement('⚠️ No se pudo verificar al jugador', player.id, 0xFF6600); return false; }
        if (CREATOR_AUTHS.indexOf(muteTargetAuth) !== -1) {
            room.sendAnnouncement('⚠️ No puedes mutear a un creador', player.id, 0xFF6600);
            return false;
        }
        var muteDurationMs = muteMinutes * 60000;
        var muteNow = Date.now();
        botState.spamMutes[muteTargetAuth] = { until: muteNow + muteDurationMs, count: (botState.spamMutes[muteTargetAuth] ? botState.spamMutes[muteTargetAuth].count : 0) + 1 };
        if (botState.spamMuteTimers[muteTargetAuth]) clearTimeout(botState.spamMuteTimers[muteTargetAuth]);
        botState.spamMuteTimers[muteTargetAuth] = setTimeout(function() {
            delete botState.spamMutes[muteTargetAuth];
            delete botState.spamMuteTimers[muteTargetAuth];
            var ps = room.getPlayerList();
            for (var mi = 0; mi < ps.length; mi++) {
                if (getAuth(ps[mi]) === muteTargetAuth) {
                    room.sendAnnouncement('🔊 Ya puedes chatear de nuevo.', ps[mi].id, 0x00FF00, 'bold');
                    break;
                }
            }
        }, muteDurationMs);
        room.sendAnnouncement('🔇 ' + player.name + ' muteó a ' + muteTarget.name + ' por ' + muteMinutes + ' minuto(s)', null, 0xFF6600, 'bold');
        room.sendAnnouncement('🔇 Estás muteado por ' + muteMinutes + ' minuto(s).', muteTargetId, 0xFF0000, 'bold');
        return false;
    }

    // !ban #ID minutos - CREADORES: banear temporalmente a un jugador
    if (message.startsWith('!ban')) {
        if (!playerAuth || CREATOR_AUTHS.indexOf(playerAuth) === -1) {
            room.sendAnnouncement('⛔ Solo los CREADORES pueden usar !ban', player.id, 0xFF0000);
            return false;
        }
        var banParts = message.split(' ');
        if (banParts.length < 3) {
            room.sendAnnouncement('⚠️ Usa: !ban #ID minutos\nEjemplo: !ban #3 5', player.id, 0xFFFF00);
            return false;
        }
        var banTargetId = parseInt(banParts[1].replace('#', ''));
        var banMinutes = parseInt(banParts[2]);

        if (isNaN(banMinutes) || banMinutes < 1) { room.sendAnnouncement('⚠️ Minutos inválidos (mínimo 1)', player.id, 0xFF6600); return false; }
        var banTarget = room.getPlayer(banTargetId);

        if (CREATOR_AUTHS.indexOf(banTargetAuth) !== -1) {
            room.sendAnnouncement('⚠️ No puedes banear a un creador', player.id, 0xFF6600);
            return false;
        }
        var banDurationSec = banMinutes * 60;
        room.sendAnnouncement('🔨 ' + player.name + ' baneó a ' + banTarget.name + ' por ' + banMinutes + ' minuto(s)', null, 0xFF0000, 'bold');
        applyTempBan(banTargetId, banDurationSec);
        return false;
    }

    // Check rifa activa: interceptar números 1-10
    if (botState.rifaActive && botState.rifaActive.started && !message.startsWith('!')) {
        var rifaNum = parseInt(message.trim());
        if (!isNaN(rifaNum) && rifaNum >= 1 && rifaNum <= 10 && playerAuth && playerAuth !== botState.rifaActive.ownerAuth) {
            // Cooldown por jugador: 3 segundos entre intentos

            if (!botState.rifaActive.cooldowns) botState.rifaActive.cooldowns = {};

                return false;
            }
            botState.rifaActive.cooldowns[playerAuth] = rifaNow;
            if (rifaNum === botState.rifaActive.number) {
                clearTimeout(botState.rifaActive.timeout);
                var rifaPot = botState.rifaActive.pot;
                try { window.__dbAddBalance(playerAuth, player.name, rifaPot); } catch(e) {}
                room.sendAnnouncement('🎉 ' + player.name + ' ADIVINÓ EL NÚMERO ' + rifaNum + '! Gana ' + formatMoney(rifaPot), null, 0x00FF00, 'bold', 2);
                botState.rifaActive = null;
            } else {
                room.sendAnnouncement('❌ ' + player.name + ' dijo ' + rifaNum + '... No es!', null, 0xFF6600);
            }
            return false;
        }
    }

    // Control de chat de Lucky (selección de jugadores)
    if (LUCKY.isActive()) {
        var result = LUCKY.onPlayerChat(player, message);
        if (result === false) return false; // Si retorna false, bloquear chat
    }
    
    // Control de chat de Lucky Hell (selección de jugadores)
    if (LUCKY_HELL.isActive()) {
        var result = LUCKY_HELL.onPlayerChat(player, message);
        if (result === false) return false; // Si retorna false, bloquear chat
    }
    
    // Control de chat de Lucky DIOS (selección de jugadores)
    if (LUCKY_DIOS.isActive()) {
        var result = LUCKY_DIOS.onPlayerChat(player, message);
        if (result === false) return false; // Si retorna false, bloquear chat

        var result = LALALA.onPlayerChat(player);
        if (result === false) return false;
    }

        var result = SURVIVAL.onPlayerChat(player);
        if (result === false) return false;

        var result = METEORS.onPlayerChat(player);

            room.sendAnnouncement('🔧 !fix voto: ' + currentVotes + '/' + needed + ' (faltan ' + (needed - currentVotes) + ')', null, 0xFFFF00);
        }
        return false;
    }

    return true;
};

room.onGameStart = function() {
    console.log("🎮 Juego iniciado");
    botState.gamePaused = false;
    // Inicializar actividad AFK de todos los jugadores en equipo
    var now = Date.now();
    var ps = room.getPlayerList();
    botState.afkLastActivity = {};
    for (var i = 0; i < ps.length; i++) {
        if (ps[i].id !== 0 && ps[i].team !== 0) {
            botState.afkLastActivity[ps[i].id] = now;
        }
    }
};

room.onGameStop = function() {
    console.log("⏹️ Juego detenido");
    botState.gamePaused = false;
};

room.onGamePause = function() {
    botState.gamePaused = true;
};

room.onGameUnpause = function() {
    botState.gamePaused = false;
    // Resetear actividad AFK al despausar (para no acumular tiempo de pausa)
    var now = Date.now();
    for (var id in botState.afkLastActivity) {
        botState.afkLastActivity[id] = now;
    }
};

// SAFETY: detectar juegos atascados sin jugadores en equipo
// ============================================
var _safetyStuckSince = null;
setInterval(function() {
    if (!botState.gameActive) { _safetyStuckSince = null; return; }
    // No interferir con juegos que ponen jugadores en espectador (es normal)
    if (LUCKY.isActive() || (LUCKY_HELL && LUCKY_HELL.isActive && LUCKY_HELL.isActive()) || (LUCKY_DIOS && LUCKY_DIOS.isActive && LUCKY_DIOS.isActive()) || (IMPOSTOR && IMPOSTOR.isActive && IMPOSTOR.isActive()) || (PISTOLERO && PISTOLERO.isActive && PISTOLERO.isActive())) {
        _safetyStuckSince = null;
        return;
    }
    var activePlayers = room.getPlayerList().filter(function(p) { return p.id !== 0 && p.team !== 0; });
    if (activePlayers.length === 0) {
        if (!_safetyStuckSince) {
            _safetyStuckSince = Date.now();
        } else if (Date.now() - _safetyStuckSince > 15000) {
            console.log('⚠️ SAFETY: juego activo sin jugadores en equipo por 15s. Forzando reinicio.');
            room.sendAnnouncement('⚠️ Juego atascado detectado. Reiniciando...', null, 0xFF6600, 'bold');
            forceStopAllGames();
            _safetyStuckSince = null;
            setTimeout(checkAndStart, 3000);
        }
    } else {
        _safetyStuckSince = null;
    }
}, 5000);

// SISTEMA AFK: auto-kick por inactividad (25 segundos sin presionar tecla)
// ============================================
room.onPlayerActivity = function(player) {
    botState.afkLastActivity[player.id] = Date.now();
};

setInterval(function() {
    // Solo verificar durante juego activo y sin pausa
    if (!botState.gameActive || botState.gamePaused) return;

    var now = Date.now();
    var players = room.getPlayerList().filter(function(p) { return p.id !== 0 && p.team !== 0; });

    players.forEach(function(p) {
        if (!botState.afkLastActivity[p.id]) {
            botState.afkLastActivity[p.id] = now;
            return;
        }

        var inactiveMs = now - botState.afkLastActivity[p.id];
        if (inactiveMs >= 25000) { // 25 segundos sin presionar tecla
            try {
                room.sendAnnouncement('⏰ ' + p.name + ' fue kickeado por AFK (25s sin presionar tecla)', null, 0xFF6600);
                room.kickPlayer(p.id, 'AFK - 25 segundos sin actividad', false);
            } catch(e) {}
            delete botState.afkLastActivity[p.id];
        }
    });

    // Limpiar entradas de jugadores que ya no están en equipo
    var activeIds = players.map(function(p) { return p.id; });
    for (var pid in botState.afkLastActivity) {
        if (activeIds.indexOf(parseInt(pid)) === -1) {
            delete botState.afkLastActivity[pid];
        }
    }
}, 5000);
